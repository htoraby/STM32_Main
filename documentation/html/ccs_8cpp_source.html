00073     osEvent event = osMessageGet(ledMessage_, 300);
00074     if (event.status == osEventMessage) {
00075       offAllLeds();
00076       condition = event.value.v;
00077       switch (condition) {
00078         case LedConditionStop:

00080           break;
00081         case LedConditionWaitApv:
00082           onLed(WaitLed);
00083           break;
00084         case LedConditionDelay:
00085           onLed(WaitLed);

00087           break;
00088         case LedConditionBlock:

00091         case LedConditionRun:
00092           onLed(WorkLed);
00093           break;
00094       }
00095     }
00096 
00097     switch (condition) {
00098       case LedConditionRunning: case LedConditionStopping:
00099         toggleLed(WorkLed);
