00028 // This is the standard default implementation for the routine to
00029 // process args. It returns a single empty arg.
00030 // For semihosting applications, this is redefined to get the real
00031 // args from the debugger. You can also use it if you decide to keep
00032 // some args in a non-volatile memory.

00034 void __attribute__((weak))
00035 __initialize_args(int* p_argc, char*** p_argv)
00036 {
00037   // By the time we reach this, the data and bss should have been initialised.
00038 
00039   // The strings pointed to by the argv array shall be modifiable by the
00040   // program, and retain their last-stored values between program startup
00041   // and program termination. (static, no const)
00042   static char name[] = "";
00043 
00044   // The string pointed to by argv[0] represents the program name;
00045   // argv[0][0] shall be the null character if the program name is not
00046   // available from the host environment. argv[argc] shall be a null pointer.
00047   // (static, no const)
00048   static char* argv[2] =
00049     { name, NULL };
00050 
00051   *p_argc = 1;
00052   *p_argv = &argv[0];

00417 void
00418 __initialize_args (int* p_argc, char*** p_argv)
00419   {
00420 
00421     // Array of chars to receive the command line from the host
00422     static char args_buf[ARGS_BUF_ARRAY_SIZE];
00423 
00424     // Array of pointers to store the final argv pointers (pointing
00425     // in the above array).
00426     static char* argv_buf[ARGV_BUF_ARRAY_SIZE];
00427 
00428     int argc = 0;
00429     int isInArgument = 0;
00430 
00431     CommandLineBlock cmdBlock;
00432     cmdBlock.pCommandLine = args_buf;
00433     cmdBlock.size = sizeof(args_buf) - 1;
00434 
00435     int ret = call_host (SEMIHOSTING_SYS_GET_CMDLINE, &cmdBlock);
00436     if (ret == 0)
00437       {
00438 
00439         // In case the host send more than we can chew, limit the
00440         // string to our buffer.
00441         args_buf[ARGS_BUF_ARRAY_SIZE - 1] = '\0';
00442 
00443         // The command line is a null terminated string
00444         char* p = cmdBlock.pCommandLine;
00445 
00446         int delim = '\0';
00447         int ch;
00448 
00449         while ((ch = *p) != '\0')
00450           {
00451             if (isInArgument == 0)
00452               {
00453                 if (!isblank(ch))
00454                   {
00455                     if (argc
00456                         >= (int) ((sizeof(argv_buf) / sizeof(argv_buf[0])) - 1))
00457                     break;
00458 
00459                     if (ch == '"' || ch == '\'')
00460                       {
00461                         // Remember the delimiter to search for the
00462                         // corresponding terminator
00463                         delim = ch;
00464                         ++p;// skip the delimiter
00465                         ch = *p;
00466                       }
00467                     // Remember the arg beginning address
00468                     argv_buf[argc++] = p;
00469                     isInArgument = 1;
00470                   }
00471               }
00472             else
00473               {
00474                 if ((ch == delim) || isblank(ch))
00475                   {
00476                     delim = '\0';
00477                     *p = '\0';
00478                     isInArgument = 0;
00479                   }
00480               }
00481             ++p;
00482           }
00483       }
00484 
00485     if (argc == 0)
00486       {
00487         // No args found in string, return a single empty name.
00488         args_buf[0] = '\0';
00489         argv_buf[0] = &args_buf[0];
00490         ++argc;
00491       }
00492 
00493     // Must end the array with a null pointer.
00494     argv_buf[argc] = NULL;
00495 
00496     *p_argc = argc;
00497     *p_argv = &argv_buf[0];
00498 
00499     // temporary here
00500     initialise_monitor_handles ();
00501 
00502     return;
00503   }

00579 /* Return a pointer to the structure associated with
00580  the user file descriptor fd. */
00581 static struct fdent*
00582 findslot (int fd)
00583   {
00584     CHECK_INIT(_REENT);
00585 
00586     /* User file descriptor is out of range. */
00587     if ((unsigned int) fd >= MAX_OPEN_FILES)
00588     return NULL;
00589 
00590     /* User file descriptor is open? */
00591     if (openfiles[fd].handle == -1)
00592     return NULL;
00593 
00594     /* Valid. */
00595     return &openfiles[fd];
00596   }

00598 /* Return the next lowest numbered free file
00599  structure, or -1 if we can't find one. */
00600 static int
00601 newslot (void)
00602   {
00603     int i;
00604 
00605     for (i = 0; i < MAX_OPEN_FILES; i++)
00606     if (openfiles[i].handle == -1)
00607     break;
00608 
00609     if (i == MAX_OPEN_FILES)
00610     return -1;

00612     return i;

00615 void
00616 initialise_monitor_handles (void)
00617   {
00618     int i;
00619 
00620     /* Open the standard file descriptors by opening the special
00621      * teletype device, ":tt", read-only to obtain a descriptor for
00622      * standard input and write-only to obtain a descriptor for standard
00623      * output. Finally, open ":tt" in append mode to obtain a descriptor
00624      * for standard error. Since this is a write mode, most kernels will
00625      * probably return the same value as for standard output, but the
00626      * kernel can differentiate the two using the mode flag and return a
00627      * different descriptor for standard error.
00628      */
00629 
00630     int volatile block[3];
00631 
00632     block[0] = (int) ":tt";
00633     block[2] = 3; /* length of filename */
00634     block[1] = 0; /* mode "r" */
00635     monitor_stdin = call_host (SEMIHOSTING_SYS_OPEN, (void*) block);
00636 
00637     block[0] = (int) ":tt";
00638     block[2] = 3; /* length of filename */
00639     block[1] = 4; /* mode "w" */
00640     monitor_stdout = call_host (SEMIHOSTING_SYS_OPEN, (void*) block);
00641 
00642     block[0] = (int) ":tt";
00643     block[2] = 3; /* length of filename */
00644     block[1] = 8; /* mode "a" */
00645     monitor_stderr = call_host (SEMIHOSTING_SYS_OPEN, (void*) block);
00646 
00647     /* If we failed to open stderr, redirect to stdout. */
00648     if (monitor_stderr == -1)
00649     monitor_stderr = monitor_stdout;
00650 
00651     for (i = 0; i < MAX_OPEN_FILES; i++)
00652     openfiles[i].handle = -1;
00653 
00654     openfiles[0].handle = monitor_stdin;
00655     openfiles[0].pos = 0;
00656     openfiles[1].handle = monitor_stdout;
00657     openfiles[1].pos = 0;
00658     openfiles[2].handle = monitor_stderr;
00659     openfiles[2].pos = 0;
00660   }

00701 /* fd, is a valid user file handle.
00702  Translates the return of _swiread into
00703  bytes read. */
00704 int
00705 _read (int fd, char* ptr, int len)
00706   {
00707     int res;
00708     struct fdent *pfd;
00709 
00710     pfd = findslot (fd);
00711     if (pfd == NULL)
00712       {
00713         errno = EBADF;
00714         return -1;
00715       }
00716 
00717     res = _swiread (pfd->handle, ptr, len);
00718 
00719     if (res == -1)
00720     return res;
00721 
00722     pfd->pos += len - res;
00723 
00724     /* res == len is not an error,
00725      at least if we want feof() to work.  */
00726     return len - res;
00727   }

00729 /* fd, is a user file descriptor. */
00730 int
00731 _swilseek (int fd, int ptr, int dir)
00732   {
00733     int res;
00734     struct fdent *pfd;
00735 
00736     /* Valid file descriptor? */
00737     pfd = findslot (fd);
00738     if (pfd == NULL)
00739       {
00740         errno = EBADF;
00741         return -1;
00742       }
00743 
00744     /* Valid whence? */
00745     if ((dir != SEEK_CUR) && (dir != SEEK_SET) && (dir != SEEK_END))
00746       {
00747         errno = EINVAL;
00748         return -1;
00749       }
00750 
00751     /* Convert SEEK_CUR to SEEK_SET */
00752     if (dir == SEEK_CUR)
00753       {
00754         ptr = pfd->pos + ptr;
00755         /* The resulting file offset would be negative. */
00756         if (ptr < 0)
00757           {
00758             errno = EINVAL;
00759             if ((pfd->pos > 0) && (ptr > 0))
00760             errno = EOVERFLOW;
00761             return -1;
00762           }
00763         dir = SEEK_SET;
00764       }
00765 
00766     int block[2];
00767     if (dir == SEEK_END)
00768       {
00769         block[0] = pfd->handle;
00770         res = checkerror (call_host (SEMIHOSTING_SYS_FLEN, block));
00771         if (res == -1)
00772         return -1;
00773         ptr += res;
00774       }
00775 
00776     /* This code only does absolute seeks.  */
00777     block[0] = pfd->handle;
00778     block[1] = ptr;
00779     res = checkerror (call_host (SEMIHOSTING_SYS_SEEK, block));
00780 
00781     /* At this point ptr is the current file position. */
00782     if (res >= 0)
00783       {
00784         pfd->pos = ptr;
00785         return ptr;
00786       }
00787     else
00788     return -1;
00789   }

00841 int
00842 _swiopen (const char* path, int flags)
00843   {
00844     int aflags = 0, fh;
00845     uint32_t block[3];
00846 
00847     int fd = newslot ();
00848 
00849     if (fd == -1)
00850       {
00851         errno = EMFILE;
00852         return -1;
00853       }
00854 
00855     /* It is an error to open a file that already exists. */
00856     if ((flags & O_CREAT) && (flags & O_EXCL))
00857       {
00858         struct stat st;
00859         int res;
00860         res = _stat (path, &st);
00861         if (res != -1)
00862           {
00863             errno = EEXIST;
00864             return -1;
00865           }
00866       }
00867 
00868     /* The flags are Unix-style, so we need to convert them. */
00869 #ifdef O_BINARY
00870     if (flags & O_BINARY)
00871     aflags |= 1;
00872 #endif
00873 
00874     /* In O_RDONLY we expect aflags == 0. */
00875 
00876     if (flags & O_RDWR)
00877     aflags |= 2;
00878 
00879     if ((flags & O_CREAT) || (flags & O_TRUNC) || (flags & O_WRONLY))
00880     aflags |= 4;
00881 
00882     if (flags & O_APPEND)
00883       {
00884         /* Can't ask for w AND a; means just 'a'.  */
00885         aflags &= ~4;
00886         aflags |= 8;
00887       }
00888 
00889     block[0] = (uint32_t) path;
00890     block[2] = strlen (path);
00891     block[1] = (uint32_t) aflags;
00892 
00893     fh = call_host (SEMIHOSTING_SYS_OPEN, block);
00894 
00895     /* Return a user file descriptor or an error. */
00896     if (fh >= 0)
00897       {
00898         openfiles[fd].handle = fh;
00899         openfiles[fd].pos = 0;
00900         return fd;
00901       }
00902     else
00903     return error (fh);
00904   }

00920 int
00921 _close (int fd)
00922   {
00923     int res;
00924     struct fdent *pfd;
00925 
00926     pfd = findslot (fd);
00927     if (pfd == NULL)
00928       {
00929         errno = EBADF;
00930         return -1;
00931       }
00932 
00933     /* Handle stderr == stdout. */
00934     if ((fd == 1 || fd == 2) && (openfiles[1].handle == openfiles[2].handle))
00935       {
00936         pfd->handle = -1;
00937         return 0;
00938       }
00939 
00940     /* Attempt to close the handle. */
00941     res = _swiclose (pfd->handle);
00942 
00943     /* Reclaim handle? */
00944     if (res == 0)
00945     pfd->handle = -1;
00946 
00947     return res;

01095 int
01096 _system (const char* s)
01097   {
01098     uint32_t block[2];
01099     int e;
01100 
01101     /* Hmmm.  The ARM debug interface specification doesn't say whether
01102      SYS_SYSTEM does the right thing with a null argument, or assign any
01103      meaning to its return value.  Try to do something reasonable....  */
01104     if (!s)
01105     return 1; /* maybe there is a shell available? we can hope. :-P */
01106     block[0] = (uint32_t) s;
01107     block[1] = strlen (s);
01108     e = checkerror (call_host (SEMIHOSTING_SYS_SYSTEM, block));
01109     if ((e >= 0) && (e < 256))
01110       {
01111         /* We have to convert e, an exit status to the encoded status of
01112          the command.  To avoid hard coding the exit status, we simply
01113          loop until we find the right position.  */
01114         int exit_code;
01115 
01116         for (exit_code = e; e && WEXITSTATUS (e) != exit_code; e <<= 1)
01117         continue;
01118       }
01119     return e;
01120   }
