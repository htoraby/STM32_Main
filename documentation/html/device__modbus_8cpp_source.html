00242 void DeviceModbus::writeModbusParameter(int ID, float Value)
00243 {
00244   int Index = getIndexAtID(ID);
00245   // Получаем всю структуру параметра
00246   ModbusParameter *Param = getFieldAll(Index);
00247   // Применяем преобразование единиц измерения
00248   Value = (Value * (Units[Param->Physic][Param->Unit][0])) + (Units[Param->Physic][Param->Unit][1]);
00249   // Применяем преобразование коэффициента
00250   Value = Value * Param->Coefficient;
00251   // Применяем преобразование масштаба
00252   Value = Value / Param->Scale;
00253   // Применяем тип данных
00254   switch (Param->TypeData) {
00255     case  TYPE_DATA_CHAR:
00256       Param->Value.tdChar[0] = (char)(Value + 0.5);
00257       break;
00258     case  TYPE_DATA_INT16:
00259       Param->Value.tdInt16[0] = (short int)(Value + 0.5);
00260       break;
00261     case  TYPE_DATA_INT32:
00262       Param->Value.tdInt32 = (int)(Value + 0.5);
00263       break;
00264     case TYPE_DATA_UINT16:
00265       Param->Value.tdUint16[0] = (unsigned short int)(Value + 0.5);
00266       break;
00267     case  TYPE_DATA_UINT32:
00268       Param->Value.tdUint32 = (unsigned int)(Value + 0.5);
00269       break;
00270     case TYPE_DATA_FLOAT:
00271       Param->Value.tdFloat = Value;
00272       break;
00273     default:
00274       break;
00275   }
00276   Param->Flag = 1;
00277   Param->Priority = 1;
00278   putMessageOutOfTurn(Index);
