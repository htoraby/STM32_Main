00065 int ModbusMasterSerial::receiveAnswer(unsigned char *Buf)
00066 {
00067   // Если истек таймаут ожидания ответа
00068   if (osSemaphoreWait(semaphoreAnswer_, MODBUS_ANSWER_TIMEOUT) == osEventTimeout) {
00069     // Возвращаем ошибку что нет ответа от устройства
00070     return 0;
00071   }
00072   // Получили первый байт
00073   else {
00074     // Крутимся пока время между байтами не станет больше MODBUS_TIME_END_PACKAGE
00075     while (1) {
00076       if (osSemaphoreWait(semaphoreAnswer_, MODBUS_TIME_END_PACKAGE) == osEventTimeout) {
00077         // Получаем количество полученных байт и массив байт
00078         return uartReadData((uartNum)numberComPort_, Buf);
00079       }
00080     }
00081   }
