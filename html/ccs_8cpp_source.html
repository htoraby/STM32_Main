00703 void Ccs::calcRegimeChangeFreqPeriodOneStep()
00704 {
00705   float beginFreq = parameters.get(CCS_RGM_CHANGE_FREQ_BEGIN_FREQ);
00706   float endFreq = parameters.get(CCS_RGM_CHANGE_FREQ_END_FREQ);

00708   float periodOneStep = 0;
00709   if ((endFreq - beginFreq) != 0)

00711   if (periodOneStep < 5)
00712     periodOneStep = 5;
00713   setValue(CCS_RGM_CHANGE_FREQ_PERIOD_ONE_STEP, periodOneStep);
