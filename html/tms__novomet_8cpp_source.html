00037 void TmsNovomet::initParameters()
00038 {
00039   Tms::initParameters();
00040   // Заполняем карту регистров ТМС
00041   int count = sizeof(modbusParameters_)/sizeof(ModbusParameter);
00042   for (int indexModbus = 0; indexModbus <= count; indexModbus++) {
00043     int indexDevice = getIndexAtId(dm_->getFieldID(indexModbus));
00044     if (indexDevice) {
00045       setFieldAccess(indexDevice, ACCESS_OPERATOR);
00046       setFieldOperation(indexDevice, dm_->getFieldOperation(indexModbus));
00047       setFieldPhysic(indexDevice, dm_->getFieldPhysic(indexModbus));
00048       float tempVal = dm_->getFieldMinimum(indexModbus);
00049       tempVal = applyCoef(tempVal, dm_->getFieldCoefficient(indexModbus));
00050       tempVal = applyUnit(tempVal, dm_->getFieldPhysic(indexModbus), dm_->getFieldUnit(indexModbus));
00051       setFieldMin(indexDevice, tempVal);
00052       tempVal = dm_->getFieldMaximum(indexModbus);
00053       tempVal = applyCoef(tempVal, dm_->getFieldCoefficient(indexModbus));
00054       tempVal = applyUnit(tempVal, dm_->getFieldPhysic(indexModbus), dm_->getFieldUnit(indexModbus));
00055       setFieldMax(indexDevice, tempVal);
00056       tempVal = dm_->getFieldDefault(indexModbus);
00057       tempVal = applyCoef(tempVal, dm_->getFieldCoefficient(indexModbus));
00058       tempVal = applyUnit(tempVal, dm_->getFieldPhysic(indexModbus), dm_->getFieldUnit(indexModbus));
00059       setFieldDef(indexDevice, tempVal);
00060       setFieldValidity(indexDevice, dm_->getFieldValidity(indexModbus));
00061       setFieldValue(indexDevice, getFieldDefault(indexDevice));
00062     }
00063   }
00064 }
