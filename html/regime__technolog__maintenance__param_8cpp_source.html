00025   switch (state_) {
00026   case IdleState:
00027     timer_ = 0;
00028     delta_ = 0;
00029     if (action_ != OffAction) { // Режим - включен
00030       if (parameters.get(CCS_CONDITION) == CCS_CONDITION_RUN) {  // Двигатель - работа;
00031         state_ = WorkState;
00032       }
00033     }
00034     break;
00035   case WorkState:
00036     {
00037       if (timer_ == 0) {
00038         timer_ = ksu.getTime();
00039       }
00040 
00041       float curValue;
00042       float setpoint;
00043       int type = parameters.get(CCS_RGM_MAINTENANCE_PARAM_TYPE);
00044       switch (type) {
00045       case PressureParam:
00046         curValue = parameters.get(TMS_PRESSURE_INTAKE);
00047         setpoint = parameters.get(CCS_RGM_MAINTENANCE_PARAM_SETPOINT_PRESSURE);
00048         break;
00049       case TemperatureParam:
00050         curValue = parameters.get(TMS_TEMPERATURE_WINDING);
00051         setpoint = parameters.get(CCS_RGM_MAINTENANCE_PARAM_SETPOINT_TEMPERATURE);
00052         break;
00053       case AnalogIn1Param: case AnalogIn2Param:
00054       case AnalogIn3Param: case AnalogIn4Param:
00055         curValue = parameters.get(CCS_AI_1_VALUE + (type - AnalogIn1Param));
00056         setpoint = parameters.get(CCS_RGM_MAINTENANCE_PARAM_SETPOINT_ANALOG_IN);
00057         break;
00058       default:
00059         curValue = parameters.get(CCS_MOTOR_CURRENT_AVARAGE);
00060         setpoint = parameters.get(CCS_RGM_MAINTENANCE_PARAM_SETPOINT_CURRENT);
00061         break;
00062       }
00063 
00064       // Накопление площади для интегрального слагаемого
00065       delta_ += curValue - setpoint;
00066 
00067       float period = parameters.get(CCS_RGM_MAINTENANCE_PARAM_PERIOD);
00068       if ((ksu.getSecFromCurTime(timer_) >= period) && (period > 0)) {
00069         // Посчёт пропорционального и интегрального слагаемого
00070         float delta = curValue - setpoint;
00071         float propSum = parameters.get(CCS_RGM_MAINTENANCE_PARAM_PROP) * delta;
00072         float intSum = parameters.get(CCS_RGM_MAINTENANCE_PARAM_INT) * delta_ / period;
00073         float freq = propSum + intSum;
00074 
00075         // Регулировка частоты
00076         float minFreq = parameters.get(CCS_RGM_MAINTENANCE_PARAM_MIN_FREQ);
00077         float maxFreq = parameters.get(CCS_RGM_MAINTENANCE_PARAM_MAX_FREQ);
00078         freq = min((int)freq, (int)(0.1 * maxFreq));
00079         if (fabs(freq) > (0.1 * maxFreq)) {
00080           if (freq > 0)
00081             freq = 0.1 * maxFreq;
00082           else
00083             freq = -0.1 * maxFreq;
00084         }
00085         freq = min((int)freq, (int)(0.1 * maxFreq));
00086 
00087         // Расчёт новой частоты с учётом изменения
00088         bool depend = parameters.get(CCS_RGM_MAINTENANCE_PARAM_DEPENDENCE);
00089         freq = parameters.get(VSD_FREQUENCY) - (depend * freq) + (!depend * freq);
00090 
00091         // Проверка частоты на допустимые границы
00092         freq = max(freq, minFreq);
00093         freq = min(freq, maxFreq);
00094 
00095         vsd->setFrequency(freq);
00096 
00097         timer_ = 0;
00098         delta_ = 0;
00099         state_ = PauseState;
00100       }
00101     }
00102     break;
00103   case PauseState:
00104     if (parameters.get(VSD_FREQUENCY) == parameters.get(VSD_FREQUENCY_NOW)) {
00105       state_ = WorkState;
00106     }
00107     break;
00108   default:
00109     state_ = IdleState;
00110     break;
00111   }
00112 
00113   parameters.set(CCS_RGM_MAINTENANCE_PARAM_STATE, state_);
00114 }
