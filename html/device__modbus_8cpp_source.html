00186 void DeviceModbus::writeModbusParameter(int id, float value)
00187 {
00188   int index = getIndexAtId(id);
00189   // Получаем всю структуру параметра
00190   ModbusParameter *param = getFieldAll(index);
00191   // Применяем преобразование единиц измерения
00192   value = (value * (Device::units[param->physic][param->unit][0])) + (Device::units[param->physic][param->unit][1]);
00193   // Применяем преобразование коэффициента
00194   value = value / param->coefficient;
00195   // Применяем тип данных
00196   switch (param->typeData) {
00197     case  TYPE_DATA_CHAR:
00198       param->value.char_t[0] = (char)(value + 0.5);
00199       break;
00200     case  TYPE_DATA_INT16:
00201       param->value.int16_t[0] = (short int)(value + 0.5);
00202       break;
00203     case  TYPE_DATA_INT32:
00204       param->value.int32_t = (int)(value + 0.5);
00205       break;
00206     case TYPE_DATA_UINT16:
00207       param->value.uint16_t[0] = (unsigned short int)(value + 0.5);
00208       break;
00209     case  TYPE_DATA_UINT32:
00210       param->value.uint32_t = (unsigned int)(value + 0.5);
