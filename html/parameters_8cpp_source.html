00029 void Parameters::task()
00030 {
00031   static int time = 0;
00032   while (1) {  
00033     if (osSemaphoreWait(semaphoreId_, 1) != osEventTimeout) {
00034       save();
00035     } else {
00036       if ((++time >= PARAMS_SAVE_TIME) || !isPowerGood()) {
00037         time = 0;
00038         save();
00039       }
00040     }
00041   }
