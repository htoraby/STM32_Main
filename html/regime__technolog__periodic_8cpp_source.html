00016 void RegimeTechnologPeriodic::processing()
00017 {
00018   action_ = parameters.get(CCS_RGM_PERIODIC_MODE);
00019   state_ = parameters.get(CCS_RGM_PERIODIC_STATE);
00020 
00021   workPeriod_ = parameters.get(CCS_RGM_PERIODIC_RUN_PERIOD);
00022   stopPeriod_ = parameters.get(CCS_RGM_PERIODIC_STOP_PERIOD);
00023   workBeginTime_ = parameters.getU32(CCS_RGM_PERIODIC_RUN_BEGIN_TIME);
00024   stopBeginTime_ = parameters.getU32(CCS_RGM_PERIODIC_STOP_BEGIN_TIME);
00025   workTimeToEnd_ = parameters.get(CCS_RGM_PERIODIC_RUN_TIME_TO_END);
00026   stopTimeToEnd_ = parameters.get(CCS_RGM_PERIODIC_STOP_TIME_TO_END);
00027 
00028   LastReasonRun runReason = (LastReasonRun)parameters.get(CCS_LAST_RUN_REASON_TMP);
00029   LastReasonStop stopReason = (LastReasonStop)parameters.get(CCS_LAST_STOP_REASON);
00030 
00031   if (action_ == OffAction) { // Режим - выключен
00032     state_ = IdleState;
00033   }
00034 
00035   switch (state_) {
00036   case IdleState:
00037     workBeginTime_ = ksu.getTime();
00038     stopBeginTime_ = ksu.getTime();
00039     workTimeToEnd_ = 0;
00040     stopTimeToEnd_ = 0;
00041     addTime_ = 0;
00042     if (action_ != OffAction) {                       // Режим - включен
00043       if (ksu.isWorkMotor() && ksu.isProgramMode()) { // Двигатель - работа; Режим - программа;
00044         state_ = WorkState;
00045       }
00046     }
00047     break;
00048   case WorkState:
00049     if (ksu.getValue(CCS_CONDITION) != CCS_CONDITION_STOP) { // Станция в работе;
00050       uint32_t time = ksu.getSecFromCurTime(workBeginTime_);
00051       workTimeToEnd_ = getTimeToEnd(workPeriod_, time);
00052       if (workTimeToEnd_ == 0) {   // Время работы истекло
00053         if (ksu.isProgramMode()) { // Режим - программа;
00054           ksu.stop(LastReasonStopProgram);
00055           state_ = WaitPauseState;
00056         }
00057       }
00058     }
00059     else { // Станция в останове
00060       if (runReason != LastReasonRunNone) { // Попытка пуска
00061         // Если причина останова "Исключение"
00062         if ((stopReason == LastReasonStopOperator) ||
00063             (stopReason == LastReasonStopRemote) ||
00064             (stopReason == LastReasonStopUnderVoltIn) ||
00065             (stopReason == LastReasonStopOverVoltIn) ||
00066             (stopReason == LastReasonStopImbalanceVoltIn) ||
00067             (stopReason == LastReasonStopNoVoltage)) {
00068           if (ksu.isProgramMode()) { // Режим - программа;
00069             uint32_t stopBeginTime = parameters.getU32(CCS_LAST_STOP_DATE_TIME); // Время остановки двигателя
00070             uint32_t workEndTime = workBeginTime_ + workPeriod_;                         // Предпологаемое время останова по программе
00071             uint32_t workTimeToEnd = workEndTime - stopBeginTime;                        // Время доработки по программе
00072             uint32_t stopTime = ksu.getSecFromCurTime(stopBeginTime);                    // Время от останова до пуска
00073             workTimeToEnd = workTimeToEnd + stopTime;                                    // Время доработки
00074             if (workTimeToEnd < (30 * 60)) { // Если время доработки меньше 30 минут
00075               stopBeginTime_ = parameters.getU32(CCS_LAST_STOP_DATE_TIME); // Время перехода в паузу фиксируем как время остановки двигателя
00076               state_ = PauseState;
00077             }
00078             else {
00079               if (workTimeToEnd < workPeriod_) {
00080                 workBeginTime_ = ksu.getSecFromCurTime(workPeriod_ - workTimeToEnd);
00081               } else {
00082                 workBeginTime_ = ksu.getTime();
00083               }
00084               state_ = RunningState;
00085             }
00086           }
00087           else {
00088             state_ = IdleState;
00089           }
00090         }
00091         else {
00092           if (ksu.isProgramMode()) {
00093             workBeginTime_ = ksu.getTime();
00094             state_ = RunningState;
00095           }
00096           else {
00097             state_ = IdleState;
00098           }
00099         }
00100       }
00101     }
00102     break;
00103   case RunningState:
00104     ksu.start(runReason);
00105     state_ = WorkState;
00106     break;
00107   case WaitPauseState:
00108     if (ksu.getValue(CCS_CONDITION) == CCS_CONDITION_STOP) { // Двигатель - останов;
00109       stopBeginTime_ = ksu.getTime();
00110       state_ = PauseState;
00111     }
00112     break;
00113   case PauseState:
00114     if (ksu.isStopMotor()) {       // Двигатель - останов;
00115       uint32_t time = ksu.getSecFromCurTime(stopBeginTime_); // Прошедшее время с начала останова
00116       stopTimeToEnd_ = getTimeToEnd(stopPeriod_ + addTime_, time);
00117       if (ksu.isProgramMode()) {   // Режим - программа;
00118         if (runReason == LastReasonRunOperator) { // Попытка пуска оператором
00119           workBeginTime_ = ksu.getTime();
00120           state_ = RunningState;
00121         } else {
00122           if (stopTimeToEnd_ == 0) { // Время паузы истекло
00123             state_ = RestartState;
00124           }
00125         }
00126       }
00127       else { // Режим программы отключен
00128         state_ = StopState;
00129       }
00130     } else {
00131       state_ = IdleState;
00132     }
00133     break;
00134   case RestartState:
00135     if (ksu.isProgramMode()) { // Режим - программа;
00136       if (ksu.isStopMotor()) {
00137         if (ksu.isPrevent()) {
00138           if (!attempt_) {                // Первая попытка запуска
00139             attempt_ = true;
00140             ksu.start(LastReasonRunProgram);
00141           }
00142         } else {
00143           addTime_ = 0;
00144           attempt_ = false;
00145           ksu.start(LastReasonRunProgram);
00146           state_ = RestartState;
00147         }
00148       } else {
00149         workBeginTime_ = ksu.getTime();
00150         state_ = WorkState;
00151       }
00152     }
00153     else {
00154       state_ = PauseState;
00155     }
00156     break;
00157   case StopState:
00158     if (ksu.isStopMotor()) {     // Двигатель - останов;
00159       uint32_t time = ksu.getSecFromCurTime(stopBeginTime_);
00160       stopTimeToEnd_ = getTimeToEnd(stopPeriod_, time);
00161       if (ksu.isProgramMode()) { // Режим - программа;
00162         addTime_ = parameters.get(CCS_TIMER_DIFFERENT_START) - stopTimeToEnd_;
00163         if (addTime_ < 0) {
00164           addTime_ = 0;
00165         }
00166         else {
00167           if (int(time - stopPeriod_) > 0)
00168             addTime_ = addTime_ + (time - stopPeriod_);
00169         }
00170         state_ = PauseState;
00171       }
00172     } else {
00173       state_ = IdleState;
00174     }
00175     break;
00176   default:
00177     state_ = IdleState;
00178     break;
00179   }
00180 
00181   parameters.set(CCS_RGM_PERIODIC_STATE, state_);
00182   parameters.set(CCS_RGM_PERIODIC_RUN_BEGIN_TIME, workBeginTime_);
00183   parameters.set(CCS_RGM_PERIODIC_STOP_BEGIN_TIME, stopBeginTime_);
00184   parameters.set(CCS_RGM_PERIODIC_RUN_TIME_TO_END, workTimeToEnd_);
00185   parameters.set(CCS_RGM_PERIODIC_STOP_TIME_TO_END, stopTimeToEnd_);
