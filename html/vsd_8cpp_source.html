00035 bool Vsd::checkVsdStatus(uint8_t bit)
00036 {
00037   if (bit < VSD_STATUS_FC_I2T_ERR)
00038     return checkBit(getValue(VSD_INVERTOR_STATUS), bit);
00039   if ((bit >= VSD_STATUS_FC_I2T_ERR) && (bit < VSD_STATUS_3))
00040     return checkBit(getValue(VSD_INVERTOR_STATUS2), bit - 16);
00041   else
00042     return false;
00043 }

00080 int Vsd::setMinFrequency(float value)
00081 {
00082   // Если записываемое значение "Минимальной частоты" больше значения
00083   // "Максимальной частоты"
00084   if (value > getValue(VSD_HIGH_LIM_SPEED_MOTOR)) {
00085     return RETURN_ERROR;                    // Возвращаем ошибку
00086   }
00087   else {
00088     // Если установили новое значение в массив параметров
00089     if (!setValue(VSD_LOW_LIM_SPEED_MOTOR, value)) {
00090       // Меняем поле минимум для уставки "Максимальной частоты"
00091       setFieldMin(getIndexAtId(VSD_HIGH_LIM_SPEED_MOTOR), value);
00092       // Меняем поле минимум для уставки "Частота"
00093       setFieldMin(getIndexAtId(VSD_FREQUENCY), value);
00094       return RETURN_OK;
00095     }
00096     return RETURN_ERROR;
00097   }
00098 }
