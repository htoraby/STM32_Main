00037 void ProtectionUnderloadMotor::getOtherSetpointProt()
00038 { 
00039   if (ksu.getValue(CCS_PROT_MOTOR_UNDERLOAD_PROGRES_RESTART_FLAG)) {
00040     float progressiveRestartDelay = ksu.getValue(CCS_PROT_MOTOR_UNDERLOAD_PROGRES_RESTART_DELAY);
00041     progressiveRestartCount_ = ksu.getValue(CCS_PROT_MOTOR_UNDERLOAD_PROGRES_RESTART_COUNT);
00042     restartDelay_ = restartDelay_ + progressiveRestartDelay * progressiveRestartCount_;
00043   }
00044   else {
00045     progressiveRestartCount_ = 0;
00046   }
00047 
00048   // Если включен режим "Периодический"
00049   if (parameters.get(CCS_RGM_CHANGE_FREQ_MODE) != Regime::OffAction) {
00050     float beginFreq_ = parameters.get(CCS_RGM_CHANGE_FREQ_BEGIN_FREQ);
00051     float endFreq_ = parameters.get(CCS_RGM_CHANGE_FREQ_END_FREQ);
00052     float beginUnderload_ = parameters.get(CCS_RGM_CHANGE_FREQ_BEGIN_UNDERLOAD);
00053     if (endFreq_ - beginFreq_) {
00054       tripSetpoint_ = beginUnderload_ + (tripSetpoint_ - beginUnderload_) *
00055           (parameters.get(VSD_FREQUENCY) - beginFreq_) / (endFreq_ - beginFreq_);
00056     }
00057   } else {
00058     if (ksu.getValue(idParam_) && (vsd->getNominalFreq() > 0))
00059       tripSetpoint_ = tripSetpoint_ * pow(vsd->getCurrentFreq(), 2) / pow(vsd->getNominalFreq(), 2);
00060   }
00061 
00062   // Если включен режим "Чередования частот"
00063   if (parameters.get(CCS_RGM_ALTERNATION_FREQ_MODE) != Regime::OffAction) {
00064     if (vsd->getCurrentFreq() == parameters.get(CCS_RGM_ALTERNATION_FREQ_FREQ_1)) {
00065       tripSetpoint_ = parameters.get(CCS_RGM_ALTERNATION_FREQ_UNDERLOAD_1);
00066     }
00067     if (vsd->getCurrentFreq() == parameters.get(CCS_RGM_ALTERNATION_FREQ_FREQ_2)) {
00068       tripSetpoint_ = parameters.get(CCS_RGM_ALTERNATION_FREQ_UNDERLOAD_2);
00069     }
00070   }
00071 }
